split(delimeter)
------------------------
>>> s="28-01-2025"
>>> x=s.split("-")
>>> print(x)
['28', '01', '2025']
>>> print("Day=",x[0])
Day= 28
>>> print("Month=",x[1])
Month= 01
>>> print("Year=",x[2])
Year= 2025
>>> s="Apple#mango#kiwi-Sberry"
>>> print(s)
Apple#mango#kiwi-Sberry
>>> y=s.split("#")
>>> print(y)
['Apple', 'mango', 'kiwi-Sberry']
>>> z=y[-1].split("-")
>>> print(z)
['kiwi', 'Sberry']
>>> y[-1]=z[0]
>>> print(y)
['Apple', 'mango', 'kiwi']
>>> y.append(z[1])
>>> print(y)
['Apple', 'mango', 'kiwi', 'Sberry']
